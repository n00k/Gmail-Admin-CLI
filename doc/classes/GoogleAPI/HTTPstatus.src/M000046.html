                def set(response)
                        @code = response.code if response.code
                        @version = response.http_version if response.http_version
                        @message = response.message if response.message
                        if response.body
                                @body = response.body
                                mat = @body.match(/reason=['"]([^'"]*)['"]/)
                                @message += " (#{mat[1]})" if mat
                        end
                end
