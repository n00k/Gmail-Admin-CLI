        def inputfile(filename)
                if @OLDINSTREAM
                        @OUSTSTREAM.puts "Input stream already redirected"
                else
                        begin
                                tfile = File.open(filename,'r')
                                @OLDINSTREAM = @INSTREAM
                                @INSTREAM = tfile
                        rescue Errno::ENOENT => e
                                @OUTSTREAM.puts "File '#{filename}' not found."     
                        end
                end
        end
