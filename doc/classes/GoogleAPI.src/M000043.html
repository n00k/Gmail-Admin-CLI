                def call_meth(obj,meth,args=[])
                        @OUTSTREAM.puts "Call method #{meth} with #{args.inspect}" if @debug > 5
                        if meth.match(/\./)
                                kid = obj.send(meth.split('.')[0])
                                return call_meth(kid,meth.match(/^[^\.]*\.(.*)$/)[1],args)
                        else
                                return obj.send(meth.to_sym,*args)
                        end
                end
